Sometimes you want two figures on the same page, use the code in \Vrefrange{fig:example-figure-0}{fig:dbl-example-figure-B}.

\begin{figure}
\caption{0th caption}\label{fig:example-figure-0}}
\end{figure}

\doublecolumnfigure{
\caption{Figure caption\label{fig:dbl-example-figure-A}}
\units{Figure units}
\includegraphics{logos/GrattanSVGLogo.pdf}
\notes{Notes}
}{
\caption{Figure caption\label{fig:dbl-example-figure-B}}
\units{Figure units}
\includegraphics{logos/GrattanSVGLogo.pdf}
\source{Source}
}
