% For two figures on one page aligned at the top
\providecommand{\doublecolumnfigureTop}[2]{
  \begin{figure*}
    \begin{minipage}[t][\textheight]{\columnwidth}
      \vspace{1pt}
      #1
    \end{minipage}
    \hfill
    \begin{minipage}[t][\textheight]{\columnwidth}
      \vspace{1pt}
      #2
    \end{minipage}
  \end{figure*}
}

% For a whole-page figure
\newenvironment{figureWhole}{
\begin{figure*}
    \begin{minipage}[t][\textheight]{\textwidth}\vspace{1pt}
}{%
    \end{minipage}
\end{figure*}
}
